- `upsample_8k_to_16k`: Convert int16 bytes to float32 via `np.frombuffer(pcm_8k, dtype=np.int16).astype(np.float32) / 32768.0`, then `soxr.resample(audio, 8000, 16000, quality="HQ")`

```python
"""Voice Activity Detection state machine for the voice pipeline.

Tracks speech onset/offset, turn completion, and barge-in detection.
Uses Silero VAD speech probability from faster-whisper (VOICE-04).
Turn detection via silence threshold (D-09, VOICE-05).
Barge-in detection during TTS playback (D-10, VOICE-06).
"""
import time
from enum import Enum
from dataclasses import dataclass, field

class PipelineState(Enum):
    LISTENING = "listening"      # VAD active, feeding STT
    SPEAKING = "speaking"        # TTS playing, barge-in monitoring
    PROCESSING = "processing"    # STT->LLM in progress

class VADEvent(Enum):
    SPEECH_START = "speech_start"
    SPEECH_CONTINUE = "speech_continue"
    SILENCE = "silence"
    TURN_COMPLETE = "turn_complete"
    BARGE_IN = "barge_in"
    NONE = "none"

@dataclass
class VADConfig:
    silence_threshold_ms: float = 700.0     # D-09: silence after speech = end of turn
    barge_in_grace_ms: float = 500.0        # Anti-pattern: grace window after TTS starts
    min_speech_duration_ms: float = 100.0   # Ignore very short speech bursts (noise)

class VADState:
    def __init__(self, config: VADConfig | None = None):
        self.config = config or VADConfig()
        self.pipeline_state = PipelineState.LISTENING
        self._speech_active = False
        self._silence_start_time: float | None = None
        self._speech_start_time: float | None = None
        self._speaking_start_time: float | None = None

    def set_pipeline_state(self, state: PipelineState) -> None:
        self.pipeline_state = state
        if state == PipelineState.SPEAKING:
            self._speaking_start_time = time.monotonic()

    def on_audio_frame(self, is_speech: bool, timestamp: float | None = None) -> VADEvent:
        ...
```

Implementation:
- `on_audio_frame` uses `time.monotonic()` if timestamp not provided
- LISTENING + is_speech + not _speech_active -> set _speech_active=True, record _speech_start_time, return SPEECH_START
- LISTENING + is_speech + _speech_active -> return SPEECH_CONTINUE
- LISTENING + not is_speech + _speech_active -> start silence timer (_silence_start_time = now)
- LISTENING + not is_speech + _speech_active + silence >= threshold -> set _speech_active=False, return TURN_COMPLETE
- SPEAKING + is_speech + grace window elapsed (>500ms since _speaking_start_time) -> return BARGE_IN
- SPEAKING + is_speech + within grace window -> return NONE (ignore)
- All other cases -> return SILENCE or NONE

```python
"""Speech-to-text engine wrapping faster-whisper.

Provides streaming transcription from 16kHz float32 audio chunks.
Uses Silero VAD (built into faster-whisper) to gate input (VOICE-01, VOICE-04).
"""
import asyncio
import numpy as np
from dataclasses import dataclass, field
import structlog

logger = structlog.get_logger()

@dataclass
class STTConfig:
    model_name: str = "distil-large-v3"
    device: str = "cpu"                  # "cuda" for GPU
    compute_type: str = "int8"           # "float16" for GPU
    language: str = "en"
    vad_filter: bool = True
    vad_min_silence_ms: int = 300
    vad_threshold: float = 0.5
    no_speech_prob_threshold: float = 0.6  # Pitfall 5: reject hallucinated segments
    min_chunk_duration_s: float = 0.5      # Don't transcribe < 500ms (Pitfall 5)
    buffer_trim_s: float = 0.5             # Keep last 0.5s of buffer for context

class STTEngine:
    """Streaming speech-to-text using faster-whisper."""

    def __init__(self, config: STTConfig | None = None):
        self.config = config or STTConfig()

    async def initialize(self) -> None:
        """Load the Whisper model. Call once at startup, not per-call."""

            compute_type=self.config.compute_type,

    async def transcribe_buffer(self, audio_16k: np.ndarray) -> list[str]:
        """Transcribe a buffer of 16kHz float32 audio.
        Returns list of text segments, filtering hallucinations via no_speech_prob.
        """
        ...
```

```python
Provides sentence-chunked streaming synthesis (VOICE-02).
Splits LLM output at sentence boundaries and synthesizes each sentence
as it arrives, enabling streaming TTS alongside streaming LLM (D-07).
"""

import re
import numpy as np
from dataclasses import dataclass
import structlog

logger = structlog.get_logger()

SENTENCE_END = re.compile(r'(?<=[.!?])\s+')

@dataclass
class TTSConfig:
    model_path: str = "kokoro-v1.0.onnx"
    voices_path: str = "voices-v1.0.bin"
    voice: str = "af_sarah"
    speed: float = 1.0
    lang: str = "en-us"
    sample_rate: int = 24000  # Kokoro outputs 24kHz

class TTSEngine:
    """Streaming text-to-speech using Kokoro-ONNX."""

    def __init__(self, config: TTSConfig | None = None):
        self.config = config or TTSConfig()

        from kokoro_onnx import Kokoro
        self._kokoro = Kokoro(self.config.model_path, self.config.voices_path)
        logger.info("tts.initialized", voice=self.config.voice)

    async def synthesize(self, text: str) -> tuple[np.ndarray, int]:
        """Synthesize a single text string to audio.
        Returns (samples_float32, sample_rate) where sample_rate is 24000.
        """
        ...

    async def synthesize_stream(self, token_queue: asyncio.Queue) -> asyncio.AsyncGenerator[tuple[np.ndarray, int], None]:
        """Yield audio chunks sentence-by-sentence from a streaming token queue.
        Reads tokens from queue, splits at sentence boundaries, synthesizes
        each complete sentence immediately while buffering the incomplete tail.
        Sentinel: None in queue signals end of input.
        """
        ...
```

Implementation for `synthesize`:
- `loop = asyncio.get_event_loop()`
- `samples, sr = await loop.run_in_executor(None, lambda: self._kokoro.create(text, voice=self.config.voice, speed=self.config.speed, lang=self.config.lang))`

Implementation for `synthesize_stream`:
- Accumulate tokens from queue into `text_buffer`
- On each token: split `text_buffer` with `SENTENCE_END.split(text_buffer)`
- For each complete sentence (all but last fragment): await `self.synthesize(sentence)` and yield result
- Keep last fragment as `text_buffer`
- On None sentinel: synthesize remaining `text_buffer` if non-empty
- Log each sentence synthesis with structlog: `logger.info("tts.sentence", text=sentence[:50], duration_ms=...)`
