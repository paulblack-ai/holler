Create `docker/freeswitch/Dockerfile`:

```dockerfile
FROM debian:12-slim

ARG SIGNALWIRE_TOKEN

# Install FreeSWITCH from SignalWire repo (requires free PAT from id.signalwire.com)
RUN apt-get update && apt-get install -y --no-install-recommends \
    wget gnupg ca-certificates curl && \
    wget --http-user=signalwire --http-password=${SIGNALWIRE_TOKEN} \
    -O /usr/share/keyrings/signalwire-freeswitch-repo.gpg \
    https://freeswitch.signalwire.com/repo/deb/debian-release/bookworm/public.gpg && \
    echo "machine freeswitch.signalwire.com login signalwire password ${SIGNALWIRE_TOKEN}" > /etc/apt/auth.conf && \
    echo "deb [signed-by=/usr/share/keyrings/signalwire-freeswitch-repo.gpg] \
    https://freeswitch.signalwire.com/repo/deb/debian-release/bookworm/ bookworm main" \
    > /etc/apt/sources.list.d/freeswitch.list && \
    apt-get update && \
    apt-get install -y --no-install-recommends \
    freeswitch \
    freeswitch-mod-commands \
    freeswitch-mod-console \
    freeswitch-mod-dptools \
    freeswitch-mod-event-socket \
    freeswitch-mod-sofia \
    freeswitch-mod-dialplan-xml \
    freeswitch-mod-loopback \
    freeswitch-mod-sndfile \
    freeswitch-mod-native-file \
    freeswitch-mod-tone-stream \
    freeswitch-mod-logfile \
    && rm -rf /var/lib/apt/lists/*

# Build mod_audio_stream from source (open-source fork: amigniter)
RUN apt-get update && apt-get install -y --no-install-recommends \
    git cmake build-essential pkg-config \
    libfreeswitch-dev libwebsockets-dev && \
    git clone https://github.com/amigniter/mod_audio_stream.git /tmp/mod_audio_stream && \
    cd /tmp/mod_audio_stream && mkdir build && cd build && \
    cmake .. -DCMAKE_BUILD_TYPE=Release && make -j$(nproc) && \
    cp mod_audio_stream.so /usr/lib/freeswitch/mod/ && \
    rm -rf /tmp/mod_audio_stream && \
    apt-get purge -y git cmake build-essential && \
    apt-get autoremove -y && \
    rm -rf /var/lib/apt/lists/*

HEALTHCHECK --interval=10s --timeout=5s --retries=3 \
    CMD fs_cli -x "status" | grep -q "UP" || exit 1

CMD ["freeswitch", "-nonat", "-c"]
```

Create `docker/docker-compose.yml`:

```yaml
services:
  freeswitch:
    build:
      context: ./freeswitch
      args:
        SIGNALWIRE_TOKEN: ${SIGNALWIRE_TOKEN}
    network_mode: "host"
    volumes:
      - ../config/freeswitch:/etc/freeswitch
      - freeswitch-logs:/var/log/freeswitch
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "fs_cli", "-x", "status"]
      interval: 10s
      timeout: 5s
      retries: 3

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 3

volumes:
  freeswitch-logs:
  redis-data:
```

Key decisions embedded:
- `network_mode: host` for FreeSWITCH per Pitfall 3 (RTP port range cannot be published)

- SIGNALWIRE_TOKEN as build arg per Pitfall 6
- mod_audio_stream built from amigniter fork per D-04 and Pitfall 1 (use open-source fork only)
- HEALTHCHECK uses `fs_cli -x "status"` to verify FreeSWITCH is UP

```xml
<include>
  <context name="default">
    <!-- Inbound calls: answer and connect to audio stream WebSocket -->
    <extension name="inbound_agent">
      <condition field="destination_number" expression="^(\d+)$">
        <action application="answer"/>
        <action application="set" data="RECORD_STEREO=true"/>
        <action application="audio_stream" data="ws://host.docker.internal:8765/voice/${uuid} mono 16k"/>
        <action application="park"/>
      </condition>
    </extension>
  </context>

  <context name="public">
    <!-- Inbound from SIP trunk routes to default context -->
    <extension name="public_to_default">
      <condition field="destination_number" expression="^(\d+)$">
        <action application="transfer" data="$1 XML default"/>
      </condition>
    </extension>
  </context>
</include>
```

- Dialplan uses `ws://host.docker.internal:8765/voice/${uuid}` for WebSocket connection to Python host
